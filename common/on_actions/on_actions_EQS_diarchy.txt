on_actions = {

	on_monthly_EQS = {
		effect = {
			log = "[GetDateText]: ROOT: [Root.GetName] on_monthly_EQS"
			if = {
				limit = {
					OR = { 
						power_balance_value = {
    						id = EQS_Princess_bop
							value < -0.85
						} 
						power_balance_value = {
    						id = EQS_Princess_bop
							value > 0.85
						}
					}
					EQS = {	EQS_twilight_sparkle = { is_political_advisor = yes	} }
				}
			country_event = EQS.advisor.200 #monthly twi event
			}
		}
	}


	#FROM - Country (owner)
	#ROOT - leader
	#FROMFROM - original owner for exiled leaders
	on_unit_leader_level_up = {
		effect = {
			if = {
				limit = { 
					FROM = { 
						tag = EQS 
						NOT = {
						 	AND = { 
								ROOT = { is_character = EQS_princess_luna }
								has_country_flag = Luna_in_training 
							}
						}
					}
				}
				unit_leader_event = EQS.flavor.1 #Champion Event
			}
		 	else_if = {
		 		limit = { 
					FROM = {
						tag = EQS
						has_country_flag = Luna_in_training
					}
				} 
		 		EQS = { clr_country_flag = Luna_in_training }
			}
		}
	}
}
